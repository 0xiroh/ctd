Ej: tipoArray(“number”, [1,2,3,4]) retorna true; tipoArray(“number”, [1,2,”hola”,,4]) retorna false. */function tipoArray(tipo, array) {    let bandera = true;    tipo = tipo.toLowerCase();    if (typeof(array[0]) != tipo) {        bandera = false;    }    if (typeof(array[1]) != tipo) {        bandera = false;    }    if (typeof(array[2]) != tipo) {        bandera = false;    }    if (typeof(array[3]) != tipo) {        bandera = false;    }    return bandera;}console.log(tipoArray("NUMBER",[1,2,3,4]));console.log(tipoArray("NUMBER",[1,2,3,"hola"]));console.log(tipoArray("string",["hi","hablalo","que dice","hola"]));console.log("\n------------------------------------------------------------\n");/* 4.
